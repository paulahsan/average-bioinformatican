Optimize threshold for classification by Hyperparameter optimization¶

write function for get maximum delta

In the current approach we consider rankValue >= 1 : highRank else lowRank

We want to optimize it for maximum differences at survival probabilty ~ 0.5

In [14]:
getMaximumDelta <- function(cutOff, df){
    
    dataToAnalyze <- df  %>%
                    mutate(RankClass = ifelse(rankValue >= cutOff,
                              "highRank", "lowRank")
                          )
    surv.res <- TCGAanalyze_survival(dataToAnalyze, clusterCol = "RankClass", filename = NULL)
    
    surv.data <- surv.res$data.survplot %>%
                    filter((surv > 0.49) & (surv < 0.51)) %>%
                    select(time, surv, type)
    survHi <- surv.data %>% filter(type == "highRank") %>%
                select(time) %>% min()
    survLo <- surv.data %>% filter(type == "lowRank") %>%
                select(time) %>% min()
    
    deltSurv <- abs(survHi-survLo)
    stopifnot(is.finite(deltSurv))
    
    return(deltSurv)
    
}

plotPartialData <- function(dfExp, RBPnames, hiRBCODE, startIndex, stopIndex){
    
    dToPlot <- dfExp  %>% 
    filter(gene_name %in% RBPnames[startIndex:stopIndex]) %>% 
    reshape2::melt(id.vars = "gene_name",
                   value.name = "log2FPKM",
                   variable.name = "bigBarcode"
                  ) %>%
    mutate(RankClass = ifelse(bigBarcode %in% hiRBCODE, "highRank", "lowRank"))
    
    
    p <- ggplot(data = dToPlot,
                aes(x=RankClass,
                    y=log2FPKM,
                    fill=RankClass
                   )) +
        geom_violin() +
        facet_wrap(~gene_name,
                   ncol = 5)
    
    savename <- paste0("violinPlotForPartialData_", 
                       startIndex, "-", stopIndex,
                       "_20220707.pdf"
                      )
    
    suppressMessages(ggsave(savename))
    message(paste("saved", savename))
    
}
